@startuml
package "Flashcard Web Application" {
  
  component "Frontend" {
    [User Interface]
    [Game Interface]
    [Set Management Interface]
    [Flashcard Management Interface]
    [API Calls]

    [User Interface] --> [API Calls]
    [Game Interface] --> [API Calls]
    [Set Management Interface] --> [API Calls]
    [Flashcard Management Interface] --> [API Calls]
  }
  
  component "Backend" {
    [Authentication Service]
    [Flashcard Management Service]
    [Public Set Service]
    [API Gateway]
  }
  
  database "Database" {
    [Users Table]
    [Flashcard Sets Table]
    [Flashcards Table]
  }
  
  [API Calls] --> [API Gateway] : REST API Calls
  [API Gateway] --> [Authentication Service] : Authenticate Users
  [API Gateway] --> [Flashcard Management Service] : CRUD Operations
  [API Gateway] --> [Public Set Service] : Read Sets

  [Authentication Service] --> [Users Table] : Validate Login Data
  [Flashcard Management Service] --> [Flashcard Sets Table] : Manage Sets
  [Flashcard Management Service] --> [Flashcards Table] : Manage Flashcards
  [Public Set Service] --> [Flashcard Sets Table] : Retrieve Public Sets

  [Flashcard Sets Table] --> [Users Table] : Owner
  [Flashcards Table] --> [Flashcard Sets Table] : Flashcard Set
  [Flashcards Table] --> [Users Table] : Owner
}
@enduml
